import random
from s3cr3t import flag, key


def encrypt(m,pub_key):
    c = []
    bin_msg = bin(bytes_to_long(m))[2:]
    n,y = key
    for i in bin_msg:
        x = random.getrandbits(100)
        if (i == '1'):
            c.append((y*pow(x,5,n))%n)
        else:
            c.append(pow(x,5,n))
    return c

Each bit is encrypted using unknown numbers, except the modulus. But, on checking the modulus, one can find that it is factorizable. Next step is to question if one can find which bit is being encrypted. For that, we need to check:

pow(ct[i],phin//5,n) == 1 #this tells that the bit encrypted is 0 else 1

Solution Script:

from Crypto.Util.number import *
n = 839647959743379757835423741637185376972991646369
p,q = 716158806799958648435521, 1172432638921544069685089
f = open('ciphertext.txt').read()
L = eval(f) #FYI: This is a dangerous function :P 
pt = ""
for i in L:
    phi = (p-1)*(q-1)
    d = (phi*inverse(5,n))%n
    if (pow(i,d,n) == 1):
        pt = pt + '0'
    else:
        pt = pt + '1'
print long_to_bytes(int(pt,2))
